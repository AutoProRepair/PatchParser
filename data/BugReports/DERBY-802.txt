               <div class="mod-content">
                <div id="description-val" class="field-ignore-highlight"> 
                 <div class="user-content-block"> 
                  <p>Grégoire Dubois on the list reported this problem. From his mail: the reproduction is attached below. <br> When statement type is set to ResultSet.TYPE_SCROLL_INSENSITIVE, outofmemory exception is thrown when reading large blobs. </p> 
                  <p>import java.sql.*;<br> import java.io.*;</p> 
                  <p>/**<br> *</p> 
                  <ul> 
                   <li>@author greg<br> */<br> public class derby_filewrite_fileread {</li> 
                  </ul> 
                  <p> private static File file = new File("/mnt/BigDisk/Clips/BabyMamaDrama-JShin.wmv");<br> private static File destinationFile = new File("/home/greg/DerbyDatabase/"+file.getName());</p> 
                  <p> /** Creates a new instance of derby_filewrite_fileread */<br> public derby_filewrite_fileread() { </p> 
                  <p> }</p> 
                  <p> public static void main(String args[]) {<br> try {<br> Class.forName("org.apache.derby.jdbc.EmbeddedDriver").newInstance();<br> Connection connection = DriverManager.getConnection ("jdbc:derby:/home/greg/DerbyDatabase/BigFileTestDB;create=true", "APP", "");<br> connection.setAutoCommit(false);</p> 
                  <p> Statement statement = connection.createStatement(ResultSet.TYPE_SCROLL_INSENSITIVE, ResultSet.CONCUR_READ_ONLY);<br> ResultSet result = statement.executeQuery("SELECT TABLENAME FROM SYS.SYSTABLES");</p> 
                  <p> // Create table if it doesn't already exists.<br> boolean exist=false;<br> while ( result.next() ) </p> { if ("db_file".equalsIgnoreCase(result.getString(1))) exist=true; } 
                  <p> if ( !exist ) </p> { System.out.println("Create table db_file."); statement.execute("CREATE TABLE db_file ("+ " name VARCHAR(40),"+ " file BLOB(2G) NOT NULL)"); connection.commit(); } 
                  <p> // Read file from disk, write on DB.<br> System.out.println("1 - Read file from disk, write on DB.");<br> PreparedStatement preparedStatement=connection.prepareStatement("INSERT INTO db_file(name,file) VALUES (?,?)");<br> FileInputStream fileInputStream = new FileInputStream(file);<br> preparedStatement.setString(1, file.getName());<br> preparedStatement.setBinaryStream(2, fileInputStream, (int)file.length()); <br> preparedStatement.execute();<br> connection.commit();<br> System.out.println("2 - END OF Read file from disk, write on DB.");</p> 
                  <p> // Read file from DB, and write on disk.<br> System.out.println("3 - Read file from DB, and write on disk.");<br> result = statement.executeQuery("SELECT file FROM db_file WHERE name='"<ins>file.getName()</ins>"'");<br> byte[] buffer = new byte <span class="error">[1024]</span>;<br> result.next();<br> BufferedInputStream inputStream=new BufferedInputStream(result.getBinaryStream(1),1024);<br> FileOutputStream outputStream = new FileOutputStream(destinationFile);<br> int readBytes = 0;<br> while (readBytes!=-1) </p> { readBytes=inputStream.read(buffer,0,buffer.length); if ( readBytes != -1 ) outputStream.write(buffer, 0, readBytes); } 
                  <p> <br> inputStream.close();<br> outputStream.close();<br> System.out.println("4 - END OF Read file from DB, and write on disk.");<br> }<br> catch (Exception e) </p> { e.printStackTrace(System.err); } 
                  <p> }<br> }</p> 
                  <p>It returns<br> 1 - Read file from disk, write on DB.<br> 2 - END OF Read file from disk, write on DB.<br> 3 - Read file from DB, and write on disk.<br> java.lang.OutOfMemoryError<br> if the file is ~10MB or more</p> 
                 </div>
