               <div class="mod-content">
                <div id="description-val" class="field-ignore-highlight"> 
                 <div class="user-content-block"> 
                  <p>Its currently just Object.toString, e.g.:</p> 
                  <p>org.apache.lucene.analysis.en.PorterStemFilter@8a32165c</p> 
                  <p>But I think we should make it more useful, to end users trying to see what their chain is doing, and to make SOPs easier when debugging:</p> 
                  <div class="code panel" style="border-width: 1px;">
                   <div class="codeContent panelContent"> 
                    <pre class="code-java">EnglishAnalyzer analyzer = <span class="code-keyword">new</span> EnglishAnalyzer(TEST_VERSION_CURRENT);
<span class="code-keyword">try</span> (TokenStream ts = analyzer.tokenStream(<span class="code-quote">"body"</span>, <span class="code-quote">"Its 2013, let's fix <span class="code-keyword">this</span> already!"</span>)) {
  ts.reset();
  <span class="code-keyword">while</span> (ts.incrementToken()) {
    <span class="code-object">System</span>.out.println(ts.toString());
  }
  ts.end();
}
</pre> 
                   </div>
                  </div> 
                  <p>Proposed output:</p>

PorterStemFilter@987b9eea term=2013,bytes=[32 30 31 33],startOffset=4,endOffset=8,positionIncrement=1,type=&lt;NUM&gt;,keyword=false
PorterStemFilter@6b5dbd1f term=let,bytes=[6c 65 74],startOffset=10,endOffset=15,positionIncrement=1,type=&lt;ALPHANUM&gt;,keyword=false

PorterStemFilter@bcd8f627 term=alreadi,bytes=[61 6c 72 65 61 64 69],startOffset=25,endOffset=32,positionIncrement=2,type=&lt;ALPHANUM&gt;,keyword=false
